            ("Inter", "Inter-Regular.ttf"),
            ("InterSB", "Inter-SemiBold.ttf"),

            else:
                logger.warning(f"PDFGuideAgent: font not found at {path}, ReportLab will use fallback")

    **PDFGuideAgent class:**

    ```python
    class PDFGuideAgent:
        def __init__(self) -> None:
            self._client = anthropic.AsyncAnthropic()
            self._model = "claude-haiku-4-5-20251001"

        async def run(self, story: Story) -> PDFGuideResult:
            """
            Generate guide content + DM keyword via LLM, render PDF, upload to GCS.
            Returns PDFGuideResult with pdf_url and dm_keyword.
            """
            guide_data = await self._generate_guide_content(story)
            pdf_bytes = self._render_pdf(guide_data)
            topic_slug = _slugify(story.topic or story.headline)
            pdf_url = await asyncio.to_thread(
                self._upload_pdf_sync, pdf_bytes, f"guides/{topic_slug}.pdf"
            )
            return PDFGuideResult(pdf_url=pdf_url, dm_keyword=guide_data["dm_keyword"])
    ```

    **`_generate_guide_content` method** — single LLM call that returns both the guide structure and the DM keyword:
    - Prompt system instruction: "You write concise educational guides for the Instagram account @techwithhareen. You must return valid JSON only."
    - User prompt: Given the story topic `{story.topic or story.headline}` and lesson steps `{story.key_stats}`, return JSON with:
      - `dm_keyword`: Single uppercase word, max 8 characters, that serves as the DM trigger keyword for this guide (e.g. "CLAUDE", "NOTION", "CHATGPT", "RUNWAY"). Must be memorable and directly related to the tool or skill being taught.
      - `intro`: 2-3 sentence intro paragraph framing why this guide is valuable.
      - `steps`: list of dicts, one per key_stat, each with `title` (the STEP N: VERB part) and `body` (2-3 sentence expansion of the step — what to actually do, not just what it means).
      - `tips`: list of 2-3 practical quick-win tips, each a single sentence.
    - Parse JSON response; if malformed, raise ValueError so caller can handle.

    **`_render_pdf` method** — ReportLab canvas to BytesIO. A4 portrait (595 x 842 points):
    - **CRITICAL ordering**: call `c.showPage()` then `c.save()` THEN `return buffer.getvalue()`. Never call `getvalue()` before `save()` — buffer is empty before save.
    - Page 1 — Cover: fill page with BG color (`_BG` as 0-1 floats: `(26/255, 26/255, 46/255)`), draw brand name "@techwithhareen" in Inter font accent color near top, draw guide title in Anton font white large, draw DM keyword CTA "DM {dm_keyword} for this guide" in accent color at bottom.
    - Page 2+ — Steps: for each step, draw title in Anton/InterSB white, draw body text in Inter white. Add a new page when content overflows (check y-cursor position; if < 80 points, add new page and reset cursor). Fill each page with BG color.
    - Final page — Tips section: "QUICK WINS" header in accent, bullet list in Inter white.
    - Use `c.setFillColorRGB(r, g, b)` for colors (values 0.0-1.0). Use `c.setFont(name, size)`. Use `c.drawString(x, y, text)` for single lines and `c.drawText()` for text objects.
    - Fall back to standard Helvetica if Anton font registration failed (check `"Anton" in pdfmetrics.getRegisteredFontNames()`).
    - Return `buffer.getvalue()` as bytes.

    **`_upload_pdf_sync` static method** — synchronous GCS upload (run via asyncio.to_thread):
    ```python
    @staticmethod
    def _upload_pdf_sync(pdf_bytes: bytes, object_name: str) -> str:
        from google.cloud import storage
        project_id = os.environ.get("GCP_PROJECT_ID")
        client = storage.Client(project=project_id)
        bucket = client.bucket(GCS_BUCKET)
        blob = bucket.blob(object_name)
        blob.upload_from_string(pdf_bytes, content_type="application/pdf")
        url = f"{GCS_BASE_URL}/{object_name}"
        logger.info(f"PDFGuideAgent: uploaded PDF to {url}")
        return url
    ```

    **`_slugify` module-level helper:**
    ```python
    def _slugify(text: str) -> str:
        """Convert topic string to safe GCS object name component."""
        slug = re.sub(r"[^a-z0-9]+", "-", text.lower()).strip("-")
        return slug[:50]  # cap length for GCS key safety
    ```
